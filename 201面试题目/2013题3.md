public class TreeSet<E> extends AbstractSet<E> implements NavigableSet<E> {
	// 不带参数的构造函数。创建一个空的TreeMap
 	public TreeSet() {
		this(new TreeMap<E,Object>());
	}
  // 带比较器的构造函数。
  public TreeSet(Comparator<? super E> comparator) {
		this(new TreeMap<E,Object>(comparator));
	}
}
